# Complete Unit Test Example

```java
@ExtendWith(
        MockitoExtension.class
)
class StudentServiceTest {

    @Mock
    private StudentRepository
            repository;

    @InjectMocks
    private StudentService
            service;

    @Test
    void shouldReturnStudent(){

        Student student =
                new Student();

        student.setId(1L);

        student.setName(
                "Ashraful"
        );

        when(
                repository.findById(1L)
        ).thenReturn(
                Optional.of(student)
        );

        Student result =
                service.getStudent(
                        1L
                );

        assertEquals(
                "Ashraful",
                result.getName()
        );
    }
}
```
